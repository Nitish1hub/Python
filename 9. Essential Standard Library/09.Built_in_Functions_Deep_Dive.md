## Introduction - Python's Built-in Arsenal

Python provides powerful built-in functions that eliminate the need for loops!

```python
result = []
for i in range(len(names)):
```

```python
result = list(enumerate(names))
```

## zip() - Combining Iterables

**Define: zip()** - Combines multiple iterables element-wise.

### Basic Usage

```python
names = ['Alice', 'Bob', 'Charlie']
ages = [25, 30, 35]

# Combine into pairs
pairs = zip(names, ages)
print(list(pairs))
# [('Alice', 25), ('Bob', 30), ('Charlie', 35)]
```

### Multiple Iterables

```python
names = ['Alice', 'Bob']
ages = [25, 30]
cities = ['NYC', 'LA']

for name, age, city in zip(names, ages, cities):
    print(f"{name} is {age} from {city}")
# Alice is 25 from NYC
# Bob is 30 from LA
```

### Unequal Lengths

```python
short = [1, 2]
long = [10, 20, 30, 40]

# Stops at shortest
print(list(zip(short, long)))
# [(1, 10), (2, 20)]

# Use itertools.zip_longest for full length
from itertools import zip_longest
print(list(zip_longest(short, long, fillvalue=0)))
# [(1, 10), (2, 20), (0, 30), (0, 40)]
```

### Creating Dictionaries

```python
keys = ['name', 'age', 'city']
values = ['Alice', 25, 'NYC']

person = dict(zip(keys, values))
print(person)
# {'name': 'Alice', 'age': 25, 'city': 'NYC'}
```

### Unzipping

```python
pairs = [('Alice', 25), ('Bob', 30), ('Charlie', 35)]

# Unzip using * operator
names, ages = zip(*pairs)
print(names)  # ('Alice', 'Bob', 'Charlie')
print(ages)   # (25, 30, 35)
```

## map() - Transforming Data

**Define: map()** - Applies function to every element.

### Basic Transformation

```python
numbers = [1, 2, 3, 4, 5]

# Square each number
squared = map(lambda x: x**2, numbers)
print(list(squared))
# [1, 4, 9, 16, 25]
```

### With Named Function

```python
def to_upper(text):
    return text.upper()

words = ['hello', 'world']
uppercase = map(to_upper, words)
print(list(uppercase))
# ['HELLO', 'WORLD']
```

### Multiple Arguments

```python
def add(x, y):
    return x + y

list1 = [1, 2, 3]
list2 = [10, 20, 30]

result = map(add, list1, list2)
print(list(result))
# [11, 22, 33]
```

### Type Conversion

```python
strings = ['1', '2', '3', '4']
numbers = list(map(int, strings))
print(numbers)  # [1, 2, 3, 4]

float_strings = ['1.5', '2.3', '3.7']
floats = list(map(float, float_strings))
print(floats)  # [1.5, 2.3, 3.7]
```

## filter() - Selecting Elements

**Define: filter()** - Selects elements where function returns True.

### Basic Filtering

```python
numbers = [1, 2, 3, 4, 5, 6]

# Get even numbers
evens = filter(lambda x: x % 2 == 0, numbers)
print(list(evens))
# [2, 4, 6]
```

### With Named Function

```python
def is_positive(n):
    return n > 0

numbers = [-2, -1, 0, 1, 2]
positive = filter(is_positive, numbers)
print(list(positive))
# [1, 2]
```

### Filtering Objects

```python
users = [
    {'name': 'Alice', 'age': 25},
    {'name': 'Bob', 'age': 17},
    {'name': 'Charlie', 'age': 30}
]

adults = filter(lambda u: u['age'] >= 18, users)
for user in adults:
    print(user['name'])
# Alice
# Charlie
```

### Use None to Filter Falsy Values

```python
values = [0, 1, '', 'hello', None, [], [1, 2], False, True]

# Remove falsy values
truthy = filter(None, values)
print(list(truthy))
# [1, 'hello', [1, 2], True]
```

## sorted() and sort() - Ordering Data

### Basic Sorting

```python
numbers = [5, 2, 8, 1, 9]

# sorted() returns new list
sorted_nums = sorted(numbers)
print(sorted_nums)  # [1, 2, 5, 8, 9]
print(numbers)      # [5, 2, 8, 1, 9] (unchanged)

# .sort() modifies in place
numbers.sort()
print(numbers)  # [1, 2, 5, 8, 9]
```

### Reverse Sorting

```python
numbers = [5, 2, 8, 1, 9]

print(sorted(numbers, reverse=True))
# [9, 8, 5, 2, 1]
```

### Custom Sort Key

```python
words = ['apple', 'pie', 'zoo', 'a']

# Sort by length
sorted_by_len = sorted(words, key=len)
print(sorted_by_len)
# ['a', 'pie', 'zoo', 'apple']

# Sort by last letter
sorted_by_last = sorted(words, key=lambda w: w[-1])
print(sorted_by_last)
# ['apple', 'pie', 'a', 'zoo']
```

### Sorting Complex Objects

```python
students = [
    {'name': 'Alice', 'grade': 85},
    {'name': 'Bob', 'grade': 92},
    {'name': 'Charlie', 'grade': 78}
]

# Sort by grade
by_grade = sorted(students, key=lambda s: s['grade'])
for student in by_grade:
    print(f"{student['name']}: {student['grade']}")
# Charlie: 78
# Alice: 85
# Bob: 92
```

### Multiple Sort Keys

```python
data = [
    ('Alice', 25, 'NYC'),
    ('Bob', 25, 'LA'),
    ('Charlie', 30, 'NYC')
]

# Sort by age, then name
sorted_data = sorted(data, key=lambda x: (x[1], x[0]))
for item in sorted_data:
    print(item)
# ('Alice', 25, 'NYC')
# ('Bob', 25, 'LA')
# ('Charlie', 30, 'NYC')
```

## all() and any() - Boolean Logic

**Define: all()** - Returns True if **all** elements are truthy.
**Define: any()** - Returns True if **any** element is truthy.

### Basic Usage

```python
# all() - Every element must be True
print(all([True, True, True]))   # True
print(all([True, False, True]))  # False
print(all([]))                   # True (empty)

# any() - At least one must be True  
print(any([False, False, True]))  # True
print(any([False, False, False])) # False
print(any([]))                    # False (empty)
```

### With Conditions

```python
numbers = [2, 4, 6, 8]

# Check if all even
if all(n % 2 == 0 for n in numbers):
    print("All numbers are even")

# Check if any negative
if any(n < 0 for n in numbers):
    print("Has negative numbers")
```

### Validation

```python
def is_valid_user(user):
    """Check if user data is valid"""
    required_fields = ['name', 'email', 'age']
    
    # All required fields must exist
    if not all(field in user for field in required_fields):
        return False
    
    # All values must be truthy
    if not all(user[field] for field in required_fields):
        return False
    
    # Age must be positive
    if user['age'] <= 0:
        return False
    
    return True

print(is_valid_user({'name': 'Alice', 'email': 'alice@example.com', 'age': 25}))
# True
```

## enumerate() - Indexed Iteration

**Define: enumerate()** - Adds counter to iterable.

### Basic Usage

```python
fruits = ['apple', 'banana', 'cherry']

for index, fruit in enumerate(fruits):
    print(f"{index}: {fruit}")
# 0: apple
# 1: banana
# 2: cherry
```

### Custom Start Index

```python
fruits = ['apple', 'banana', 'cherry']

for index, fruit in enumerate(fruits, start=1):
    print(f"{index}. {fruit}")
# 1. apple
# 2. banana
# 3. cherry
```

### Finding Index

```python
names = ['Alice', 'Bob', 'Charlie', 'David']

for i, name in enumerate(names):
    if name == 'Charlie':
        print(f"Found at index {i}")
        break
# Found at index 2
```

## sum(), min(), max()

### sum()

```python
numbers = [1, 2, 3, 4, 5]
print(sum(numbers))  # 15

# With start value
print(sum(numbers, 100))  # 115

# Sum of specific values
orders = [{'amount': 10}, {'amount': 20}, {'amount': 30}]
total = sum(order['amount'] for order in orders)
print(total)  # 60
```

### min() and max()

```python
numbers = [5, 2, 8, 1, 9]
print(min(numbers))  # 1
print(max(numbers))  # 9

# With key
words = ['apple', 'pie', 'zoo']
print(min(words, key=len))  # 'pie'
print(max(words, key=len))  # 'apple'

# With objects
students = [
    {'name': 'Alice', 'grade': 85},
    {'name': 'Bob', 'grade': 92}
]
top_student = max(students, key=lambda s: s['grade'])
print(top_student)  # {'name': 'Bob', 'grade': 92}
```

## reversed() - Reverse Iteration

```python
numbers = [1, 2, 3, 4, 5]

# Iterate in reverse
for n in reversed(numbers):
    print(n)
# 5, 4, 3, 2, 1

# Convert to list
reversed_list = list(reversed(numbers))
print(reversed_list)  # [5, 4, 3, 2, 1]
```

## Real-Life Practical Examples

### Example: Data Processing Pipeline

```python
data = [
    {'name': 'Alice', 'score': 85, 'active': True},
    {'name': 'Bob', 'score': 92, 'active': False},
    {'name': 'Charlie', 'score': 78, 'active': True},
]

# Filter active users
active_users = list(filter(lambda u: u['active'], data))

# Extract scores
scores = list(map(lambda u: u['score'], active_users))

# Calculate average
average = sum(scores) / len(scores)
print(f"Average: {average}")
```
